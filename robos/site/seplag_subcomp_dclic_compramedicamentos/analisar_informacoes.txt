        Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 0 Column: $'''AE''' Row: linhaatual

        ELSE
            Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 18 Column: $'''AE''' Row: linhaatual
        END
    END
ELSE
    # NÃO É  mineiro, verifica CONFAZ
    IF analisar[0][4] = $'''SIM''' THEN
        # Se tem confaz zera as duas colunas
        Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 0 Column: $'''AF''' Row: linhaatual
        Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 0 Column: $'''AE''' Row: linhaatual
    ELSE
        # Verificar se é genérico
        IF analisar[0][8] = $'''Genérico''' THEN
            Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 12 Column: $'''AE''' Row: linhaatual
            Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 12 Column: $'''AF''' Row: linhaatual
        ELSE
            Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 18 Column: $'''AE''' Row: linhaatual
            Excel.WriteToExcel.WriteCell Instance: planilha_base Value: 18 Column: $'''AF''' Row: linhaatual
        END
    END
