**REGION Variaveis data/mes do processo. Tratamento do excel
Excel.ReadFromExcel.ReadCell Instance: planilha_base StartColumn: $'''L''' StartRow: 2 GetCellContentsMode: Excel.GetCellContentsMode.TypedValues CellValue=> ExcelDatames
Text.SplitText.SplitWithDelimiter Text: ExcelDatames CustomDelimiter: $'''/''' IsRegEx: False Result=> PartesMesAno
Text.ChangeCase Text: PartesMesAno[0] NewCase: Text.CaseOption.LowerCase Result=> MesTxt
SET AnoCompleto TO PartesMesAno[1]
SWITCH MesTxt
    CASE = $'''jan'''
        SET MesNum TO $'''01
'''
    CASE = $'''fev'''
        SET MesNum TO $'''02
'''
    CASE = $'''mar'''
        SET MesNum TO $'''03
'''
    CASE = $'''abr'''
        SET MesNum TO $'''04'''
    CASE = $'''mai'''
        SET MesNum TO $'''05
'''
    CASE = $'''jun'''
        SET MesNum TO $'''06'''
    CASE = $'''jul'''
        SET MesNum TO $'''07'''
    CASE = $'''ago'''
        SET MesNum TO $'''08'''
    CASE = $'''set'''
        SET MesNum TO $'''09

'''
    CASE = $'''out'''
        SET MesNum TO 10
    CASE = $'''nov'''
        SET MesNum TO 11
    CASE = $'''dez'''
        SET MesNum TO 12
    DEFAULT
        SET MesNum TO $'''01'''
END
SET DataIniTexto TO $'''01/%MesNum%/%AnoCompleto%'''
SET DataFimTexto TO $'''30/%MesNum%/%AnoCompleto%'''
SET pDataIni TO DataIniTexto
SET pDataFim TO DataFimTexto
**ENDREGION
